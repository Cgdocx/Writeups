```python
from Crypto.Util.number import bytes_to_long
from collections import namedtuple
from hashlib import sha256

from secret import FLAG


# Standard elliptic curve operations
Point = namedtuple("Point", "x y")
O = 'Infinity'

def check_point(P):
    if P == O:
        return True
    else:
        return (P.y**2 - (P.x**3 + a*P.x + b)) % p == 0 and 0 <= P.x < p and 0 <= P.y < p


def point_inverse(P):
    if P == O:
        return P
    return Point(P.x, -P.y % p)


def point_addition(P, Q):
    if P == O:
        return Q
    elif Q == O:
        return P
    elif Q == point_inverse(P):
        return O
    else:
        if P == Q:
            lam = (3*P.x**2 + a)*pow(2*P.y, -1, p)
            lam %= p
        else:
            lam = (Q.y - P.y) * pow((Q.x - P.x), -1, p)
            lam %= p
    Rx = (lam**2 - P.x - Q.x) % p
    Ry = (lam*(P.x - Rx) - P.y) % p
    R = Point(Rx, Ry)
    assert check_point(R)
    return R


def point_multiplication(P, n):
    Q = P
    R = O
    while n > 0:
        if n % 2 == 1:
            R = point_addition(R, Q)
        Q = point_addition(Q, Q)
        n = n // 2
    assert check_point(R)
    return R


# Standard ECDSA operations
def number_to_string(num):
    l = p.bit_length() // 8
    fmt_str = "%0" + str(2 * l) + "x"
    string = binascii.unhexlify((fmt_str % num).encode())
    assert len(string) == l, (len(string), l)
    return string.hex()


def string_to_number(string):
    return int(binascii.hexlify(string), 16)


def hash_message(message):
    return bytes_to_long(sha256(message.encode()).digest())


def sign(private_key, G, message, k=None):
    if k is None:
        k = random.randrange(n)
    hash = hash_message(message)
    r = point_multiplication(G, k).x % n
    s = pow(k, -1, n) * (hash + r * private_key) % n
    return number_to_string(r) + number_to_string(s)


def verify(public_key, G, message, signature):
    signature = bytes.fromhex(signature)
    l = p.bit_length() // 8
    if len(signature) != 2*l:
        return False
    r = string_to_number(signature[:l])
    s = string_to_number(signature[l:])
    if r < 1 or r > n - 1 or s < 1 or s > n-1:
        return False
    hash = hash_message(message)
    u1 = (hash * pow(s, -1, n)) % n
    u2 = (r * pow(s, -1, n)) % n
    P = point_addition(
        point_multiplication(G, u1), 
        point_multiplication(public_key, u2)
        )
    return P.x % n == r
```

## Solution

We are given a service implementing ECDSA signing and verification.

We can:

 * Sign any message
 * Verify any message/signature pair
 * Get the flag if we produce a new valid signature on `"can i haz flag?"`

However, if the signature was previously generated by the server, it is rejected:

```python
if signature in SIGNATURES_CACHE:
    send("I already knew that...")
```

Therefore, the goal is to obtain a valid ECDSA signature on the required string,
but not one the server has already issued.

The server implements ECDSA correctly, but it makes two critical mistakes.

First, it lets us sign the target message directly:

```python
signature = sign(private_key, G, message)
SIGNATURES_CACHE.add(signature)
```

There is no restriction preventing us from calling `sign("can i haz flag?")` and obtaining
a valid and legitimate signature for the target string (although it is cached).

And second, it only caches the exact string of the signature as 
`number_to_string(r) + number_to_string(s)` for the `(r, s)` ECDSA signature pair.
However, if we submit a different string - even if it is a valid signature for the same message - it will pass the cache check.

ECDSA has a well-known property: If `(r, s)` is a valid signature for message `m`, then `(r, n − s)` is also a valid signature.

This means for any valid signature, there is always another valid signature on the same message.

So, basically we'll ask the server to sign the target message, receive `(r, s)` from the server,
compute a fresh signature using `s' = (n − s) mod n`, and submit `(r, s')` to get the flag.

Script:

```python
from pwn import remote
import binascii

# Curve order n from the challenge
n = [AWS_SECRET_REMOVED]ada7179e84f3b9cac2fc632551

FLAG_MSG = "can i haz flag?"

def sig_to_rs(sig_hex):
    sig_bytes = bytes.fromhex(sig_hex)
    L = len(sig_bytes) // 2
    r = int.from_bytes(sig_bytes[:L], "big")
    s = int.from_bytes(sig_bytes[L:], "big")
    return r, s

def rs_to_sig(r, s):
    # number_to_string uses l = p.bit_length() // 8 = 32 bytes for P-256
    L = 32
    return r.to_bytes(L, "big").hex() + s.to_bytes(L, "big").hex()

def main():
    io = remote("0.cloud.chals.io", 27412)

    # 1) Ask for a signature on the *flag* message
    io.sendlineafter(b"3 - Exit", b"1")
    io.sendlineafter(b"Enter a message to sign:", FLAG_MSG.encode())

    line = io.recvline_contains(b"Signature:").decode().strip()
    sig_hex = line.split("Signature: ", 1)[1]
    print("[+] Got original signature:", sig_hex)

    # 2) Parse (r, s)
    r, s = sig_to_rs(sig_hex)
    print("[+] r =", hex(r))
    print("[+] s =", hex(s))

    # 3) Compute malleated s' = n - s mod n
    s2 = (n - s) % n
    print("[+] s' =", hex(s2))

    # 4) Build new signature hex
    forged_sig = rs_to_sig(r, s2)
    print("[+] Forged signature:", forged_sig)

    # 5) Verify with the same message and new signature
    io.sendlineafter(b"3 - Exit", b"2")
    io.sendlineafter(b"What is your message?", FLAG_MSG.encode())
    io.sendlineafter(b"What is your signature?", forged_sig.encode())

    # 6) Print server response
    print(io.recvline().decode())
    print(io.recvline().decode())

    io.sendlineafter(b"3 - Exit", b"3")

    print(io.recvall().decode())

if __name__ == "__main__":
    main()
```
